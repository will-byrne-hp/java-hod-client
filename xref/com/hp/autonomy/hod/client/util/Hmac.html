31      /**
32       * Generate the token header for an HMAC signed request to Haven OnDemand.
33       * @param request The request to authenticate
34       * @param token The HMAC SHA1 authentication token
35       * @return The token parameter for the request
36       */
37      public <Q, B> String generateToken(final Request<Q, B> request, final AuthenticationToken token) {
38          final String bodyHash = createBodyHash(request.getBody());
39          final String message = createMessage(request, bodyHash);
40          final String signature = base64EncodeForUri(hmacSha1(message, token.getSecret()));
41  
42          final List<String> components = Arrays.asList(
43                  token.getType(),
44                  token.getId(),
45                  bodyHash,
46                  signature
47          );
48  
49          return StringUtils.join(components, COLON);
50      }

52      // Creates the representation of the request for HMAC signing, given a request and it's body hash
53      private <Q, B> String createMessage(final Request<Q, B> request, final String bodyHash) {
54          final List<String> components = new LinkedList<>();
55          components.add(encodeVerb(request.getVerb()));
56          components.add(encodePath(request.getPath()));
57          components.addAll(encodeQueryParameters(request.getQueryParameters()));
58          components.add(urlEncode(bodyHash));
59          return StringUtils.join(components, NEW_LINE);
60      }

62      private <T> List<String> encodeQueryParameters(final Map<String, List<T>> queryParameters) {
63          if (queryParameters == null || queryParameters.isEmpty()) {
64              return Collections.emptyList();
65          } else {
66              return encodeAndSpreadParameterMap(queryParameters, new ValueEncoder() {
67                  @Override
68                  public String encode(final Object input) {
69                      return urlEncode(input.toString());
70                  }
71              });
72          }
73      }

75      private <T> String createBodyHash(final Map<String, List<T>> body) {
76          if (body == null || body.isEmpty()) {
77              // If no body, the body hash must be the empty string
78              return EMPTY;
79          } else {
80              final List<String> components = encodeAndSpreadParameterMap(body, new ValueEncoder() {
81                  @Override
82                  public String encode(final Object input) {
83                      final byte[] bytes;
84  
85                      if (input instanceof byte[]) {
86                          bytes = (byte[]) input;
87                      } else {
88                          bytes = bytesFromString(input.toString());
89                      }
90  
91                      return Hex.encodeHexString(md5Hash(bytes));
92                  }
93              });
94  
95              final String bodyRepresentation = StringUtils.join(components, NEW_LINE);
96              return base64EncodeForUri(md5Hash(bytesFromString(bodyRepresentation)));
97          }
98      }

100     /*
101         Takes a map of parameter name to list of parameter values. URI encodes every key and encodes every value using
102         the ValueEncoder, then returns a list containing each encoded value adjacent to it's encoded key. The list is
103         sorted by the encoded key, but maintains the order of each value associated with a given key.
104 
105         {key1: [value11, value12], key2: [value21]}
106         uri(key2) < uri(key1)
107 
108         => [uri(key2), encode(value21), uri(key1), encode(value11), uri(key1), encode(value12)]
109     */
110     private <T> List<String> encodeAndSpreadParameterMap(final Map<String, List<T>> parameterMap, final ValueEncoder encoder) {
111         final List<Parameter> parameters = new LinkedList<>();
112 
113         for (final Map.Entry<String, List<T>> entry : parameterMap.entrySet()) {
114             final String encodedKey = urlEncode(entry.getKey());
115             final List<?> values = entry.getValue();
116 
117             for (final Object value : values) {
118                 parameters.add(new Parameter(encodedKey, encoder.encode(value)));
119             }
120         }
121 
122         // Sort is guaranteed to be stable, so parameters for a given key will remain in the same order
123         Collections.sort(parameters);
124 
125         final List<String> components = new LinkedList<>();
126 
127         for (final Parameter parameter : parameters) {
128             components.add(parameter.key);
129             components.add(parameter.value);
130         }
131 
132         return components;
133     }

144     private byte[] bytesFromString(final String input) {
145         try {
146             return input.getBytes(UTF8);

153     private String base64EncodeForUri(final byte[] bytes) {
154         // Some base64 characters are not valid in a URI
155         return Base64.encodeBase64String(bytes).replaceAll("=", EMPTY).replaceAll("/", "-").replaceAll("[+]", "_");
156     }

159         try {
160             return URLEncoder.encode(input, UTF8);

167     private byte[] md5Hash(final byte[] input) {
168         try {
169             return MessageDigest.getInstance(MD5).digest(input);
170         } catch (final NoSuchAlgorithmException e) {
171             // This should never happen on a sensible JVM
172             throw new AssertionError("UTF8 or MD5 is not supported");
173         }
174     }

176     private byte[] hmacSha1(final String message, final String secret) {
177         try {
178             final Mac mac = Mac.getInstance(HMAC_SHA1);
179             final Key key = new SecretKeySpec(bytesFromString(secret), HMAC_SHA1);
180             mac.init(key);
181             return mac.doFinal(bytesFromString(message));
182         } catch (final NoSuchAlgorithmException e) {
183             // This should never happen on a sensible JVM
184             throw new AssertionError("HMAC SHA1 is not supported", e);
185         } catch (final InvalidKeyException e) {
186             // In practice, this means that the token secret was invalid
187             throw new IllegalArgumentException("Invalid token secret", e);
188         }
189     }
