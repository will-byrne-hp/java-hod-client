9   import com.hp.autonomy.hod.client.config.HodServiceConfig;

22  public class JobServiceImpl<T extends JobStatus<?>> implements JobService<T> {
23  
24      private final Class<? extends T> returnType;
25      private final JobBackend jobBackend;
26      private final Requester requester;
27  
28      public JobServiceImpl(final HodServiceConfig hodServiceConfig, final Class<? extends T> returnType) {
29          this.returnType = returnType;
30  
31          jobBackend = hodServiceConfig.getRestAdapter().create(JobBackend.class);
32          requester = hodServiceConfig.getRequester();
33      }
34  
35      @Override
36      public T getJobStatus(final JobId jobId) throws HodErrorException {
37          return requester.makeRequest(returnType, getStatusBackendCaller(jobId));
38      }
39  
40      @Override
41      public T getJobStatus(final TokenProxy tokenProxy, final JobId jobId) throws HodErrorException {
42          return requester.makeRequest(tokenProxy, returnType, getStatusBackendCaller(jobId));
43      }
44  
45      @Override
46      public T getJobResult(final JobId jobId) throws HodErrorException {
47          return requester.makeRequest(returnType, getResultBackendCaller(jobId));
48      }
49  
50      @Override
51      public T getJobResult(final TokenProxy tokenProxy, final JobId jobId) throws HodErrorException {
52          return requester.makeRequest(tokenProxy, returnType, getResultBackendCaller(jobId));
53      }
54  
55      private Requester.BackendCaller getStatusBackendCaller(final JobId jobId) {
56          return new Requester.BackendCaller() {
57              @Override
58              public Response makeRequest(final AuthenticationToken authenticationToken) throws HodErrorException {
59                  return jobBackend.getJobStatus(authenticationToken, jobId);
60              }
61          };
62      }
63  
64      private Requester.BackendCaller getResultBackendCaller(final JobId jobId) {
65          return new Requester.BackendCaller() {
66              @Override
67              public Response makeRequest(final AuthenticationToken authenticationToken) throws HodErrorException {
68                  return jobBackend.getJobResult(authenticationToken, jobId);
69              }
70          };
71      }
72  }
