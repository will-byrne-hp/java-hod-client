36      static SignedRequest sign(final Hmac hmac, final String endpoint, final AuthenticationToken token, final Request<String, String> request) {
37          final URIBuilder uriBuilder;
38  
39          try {
40              uriBuilder = new URIBuilder(endpoint + request.getPath());
41          } catch (final URISyntaxException e) {
42              throw new IllegalArgumentException("Invalid endpoint or request path");
43          }
44  
45          if (request.getQueryParameters() != null) {
46              for (final Map.Entry<String, List<String>> entry : request.getQueryParameters().entrySet()) {
47                  for (final String value : entry.getValue()) {
48                      uriBuilder.addParameter(entry.getKey(), value);
49                  }
50              }
51          }
52  
53          final String bodyString;
54  
55          if (request.getBody() == null) {
56              bodyString = null;
57          } else {
58              final List<NameValuePair> pairs = new LinkedList<>();
59  
60              for (final Map.Entry<String, List<String>> entry : request.getBody().entrySet()) {
61                  for (final String value: entry.getValue()) {
62                      pairs.add(new BasicNameValuePair(entry.getKey(), value));
63                  }
64              }
65  
66              bodyString = URLEncodedUtils.format(pairs, UTF8);
67          }
68  
69          final String tokenString = hmac.generateToken(request, token);
70          return new SignedRequest(uriBuilder.toString(), request.getVerb(), bodyString, tokenString);
71      }
