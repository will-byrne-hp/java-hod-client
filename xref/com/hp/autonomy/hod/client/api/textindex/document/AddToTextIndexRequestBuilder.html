18  @Setter
19  @Accessors(chain = true)
20  public class AddToTextIndexRequestBuilder {
21  
22      /**
23       * @param duplicateMode The duplicate mode of the document
24       */
25      private DuplicateMode duplicateMode;
26

28  
29      private List<Object> additionalMetadata = new ArrayList<>();
30  
31      /**

33       * @param referencePrefix The first index
34       * @param referencePrefixes The remaining referencePrefixes
35       * @return this

37      public AddToTextIndexRequestBuilder addReferencePrefixes(final String referencePrefix, final String... referencePrefixes) {
38          this.referencePrefixes.add(referencePrefix);
39          this.referencePrefixes.addAll(Arrays.asList(referencePrefixes));

41          return this;
42      }
43  
44      /**
45       * Sets the value of the referencePrefixes parameter
46       * @param referencePrefixes The referencePrefixes to query
47       * @return this
48       */
49      public AddToTextIndexRequestBuilder setReferencePrefixes(final List<String> referencePrefixes) {
50          this.referencePrefixes = referencePrefixes;
51  
52          return this;
53      }
54  
55      /**
56       * Sets the value of the metadata parameter
57       * @param metadatum The first index
58       * @param metadata The remaining metadata
59       * @return this
60       */
61      public AddToTextIndexRequestBuilder addAdditionalMetadata(final Object metadatum, final Object... metadata) {
62          this.additionalMetadata.add(metadatum);
63          this.additionalMetadata.addAll(Arrays.asList(metadata));
64  
65          return this;
66      }
67  
68      /**
69       * Sets the value of the metadata parameter
70       * @param metadata The metadata to query
71       * @return this
72       */
73      public AddToTextIndexRequestBuilder setAdditionalMetadata(final List<Object> metadata) {
74          this.additionalMetadata = metadata;
75  
76          return this;
77      }
78  
79      /**
80       * @return A map of query parameters suitable for use with {@link AddToTextIndexBackend}. get is NOT supported on
81       * the resulting map
82       */
83      Map<String, Object> build() {
84          final Map<String, Object> params = new MultiMap<>();
85  
86          params.put("duplicate_mode", duplicateMode);
87  
88          for (final String referencePrefix : referencePrefixes) {
89              params.put("reference_prefix", referencePrefix);
90          }
91  
92          for (final Object metadata : additionalMetadata) {
93              params.put("additional_metadata", metadata);
94          }
95  
96          return params;
97      }
