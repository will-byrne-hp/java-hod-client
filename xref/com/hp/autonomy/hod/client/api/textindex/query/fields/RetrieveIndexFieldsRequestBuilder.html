6   package com.hp.autonomy.hod.client.api.textindex.query.fields;
7   
8   import com.hp.autonomy.hod.client.util.MultiMap;
9   import lombok.Setter;
10  import lombok.experimental.Accessors;
11  
12  import java.util.Map;
13  
14  /**
15   * Helper class for building up optional parameters for the Retrieve Index Fields API. The default value for all parameters
16   * is null. Null parameters will not be sent to HP Haven OnDemand
17   */
18  @Setter
19  @Accessors(chain = true)
20  public class RetrieveIndexFieldsRequestBuilder {
21  
22  
23      /**
24       * @param groupFieldsByType Whether the fields should be shown grouped by field types.
25       */
26      private Boolean groupFieldsByType;
27  
28      /**
29       * @param fieldType Describe the field type of field names to retrieve.
30       */
31      private FieldType fieldType;
32  
33      /**
34       * @param maxValues The number of field names to display. Displays maximum 1000 tags by default.
35       */
36      private Integer maxValues;
37  
38  
39      /**
40       * @return A map of query parameters suitable for use with {@link RetrieveIndexFieldsBackend}. get is NOT supported on
41       * the resulting map
42       */
43      Map<String, Object> build() {
44          final Map<String, Object> map = new MultiMap<>();
45          map.put("group_fields_by_type", groupFieldsByType);
46          map.put("fieldtype", fieldType);
47          map.put("max_values", maxValues);
48  
49          return map;
50      }
51  
52  }
