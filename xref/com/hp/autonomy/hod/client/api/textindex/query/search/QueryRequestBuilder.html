165     /**

169     Map<String, Object> build() {
170         final Map<String, Object> map = new MultiMap<>();
171         map.put("end_tag", endTag);
172         map.put("field_text", fieldText);
173         map.put("highlight", highlight);
174         map.put("absolute_max_results", absoluteMaxResults);
175         map.put("max_page_results", maxPageResults);
176         map.put("min_score", minScore);
177         map.put("print", print);
178         map.put("print_fields", StringUtils.join(printFields, ','));
179         map.put("sort", sort);
180         map.put("start", start);
181         map.put("start_tag", startTag);
182         map.put("summary", summary);
183         map.put("total_results", totalResults);
184         map.put("query_profile", queryProfile);
185         map.put("promotion", promotions);
186 
187         // prefer the DateTime over the numeric versions
188         map.putAll(TimeSelector.max(maxDate, maxDateDays, maxDateSeconds));
189         map.putAll(TimeSelector.min(minDate, minDateDays, minDateSeconds));
190 
191         for (final ResourceIdentifier index : indexes) {
192             map.put("indexes", index);
193         }
194 
195         return map;
196     }
