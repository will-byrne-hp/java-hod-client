8   import com.hp.autonomy.hod.client.api.authentication.AuthenticationToken;
9   import com.hp.autonomy.hod.client.error.HodErrorException;
10  import retrofit.client.Response;
11  import retrofit.http.GET;
12  import retrofit.http.Header;
13  import retrofit.http.Multipart;
14  import retrofit.http.POST;
15  import retrofit.http.Part;
16  import retrofit.http.PartMap;
17  import retrofit.http.Query;
18  import retrofit.http.QueryMap;
19  import retrofit.mime.TypedOutput;
20  
21  import java.util.Map;
22  
23  /**
24   * Interface representing the FindSimilar API. Use the same parameters as QueryTextIndex, so does not have its own
25   * request builder
26   */
27  interface FindSimilarBackend {
28  
29      String URL = "/2/api/sync/textindex/query/findsimilar/v1";
30  
31      /**
32       * Finds similar documents to the given text using the given token
33       * @param token The token to use to authenticate the request
34       * @param text The query text
35       * @param params Additional parameters to be sent as part of the request
36       * @return A list of documents that are similar to the query text
37       */
38      @GET(URL)
39      Response findSimilarDocumentsToText(
40          @Header("token") AuthenticationToken token,
41          @Query("text") String text,
42          @QueryMap Map<String, Object> params
43      ) throws HodErrorException;
44  
45      /**
46       * Finds similar documents to the text from an object store object using the given token
47       * @param token The token to use to authenticate the request
48       * @param reference An HP Haven OnDemand reference obtained from either the Expand Container or Store Object API.
49       * The contents of the object will be used as the query text
50       * @param params Additional parameters to be sent as part of the request
51       * @return A list of documents that are similar to the query text
52       */
53      @GET(URL)
54      Response findSimilarDocumentsToReference(
55          @Header("token") AuthenticationToken token,
56          @Query("reference") String reference,
57          @QueryMap Map<String, Object> params
58      ) throws HodErrorException;
59  
60      /**
61       * Finds similar documents to the text from a url using the given token
62       * @param token The token to use to authenticate the request
63       * @param url A publicly accessible HTTP URL from which the query text can be retrieved
64       * @param params Additional parameters to be sent as part of the request
65       * @return A list of documents that are similar to the query text
66       */
67      @GET(URL)
68      Response findSimilarDocumentsToUrl(
69          @Header("token") AuthenticationToken token,
70          @Query("url") String url,
71          @QueryMap Map<String, Object> params
72      ) throws HodErrorException;
73  
74      /**
75       * Finds similar documents to the text from a url using the given token
76       * @param token The token to use to authenticate the request
77       * @param indexReference The reference of a document in HP Haven OnDemand
78       * @param params Additional parameters to be sent as part of the request
79       * @return A list of documents that are similar to the query text
80       */
81      @GET(URL)
82      Response findSimilarDocumentsToIndexReference(
83          @Header("token") AuthenticationToken token,
84          @Query("index_reference") String indexReference,
85          @QueryMap Map<String, Object> params
86      ) throws HodErrorException;
87  
88      /**
89       * Finds similar documents to the text in a file using the given token
90       * @param token The token to use to authenticate the request
91       * @param file A file containing the query text
92       * @param params Additional parameters to be sent as part of the request
93       * @return A list of documents that are similar to the query text
94       */
95      @Multipart
96      @POST(URL)
97      Response findSimilarDocumentsToFile(
98          @Header("token") AuthenticationToken token,
99          @Part("file") TypedOutput file,
100         @PartMap Map<String, Object> params
101     ) throws HodErrorException;
102 }
