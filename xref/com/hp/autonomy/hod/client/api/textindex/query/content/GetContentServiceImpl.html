19  /**
20   * Default implementation of GetContentService
21   * <p/>
22   * If a custom return type is not required, use the {@link #documentsService(HodServiceConfig)} static factory method
23   * @param <T> The desired return type of the services methods
24   */
25  public class GetContentServiceImpl<T> implements GetContentService<T> {
26      
27      private final GetContentBackend getContentBackend;
28      private final Requester requester;
29      private final Class<T> returnType;
30  
31      /**
32       * Create a new GetContentServiceImpl of the desired parameterized type
33       * @param hodServiceConfig The configuration to use
34       * @param returnType The desired return type for service methods. This type must have the correct Jackson annotations
35       * to read HP Haven OnDemand responses
36       */
37      public GetContentServiceImpl(final HodServiceConfig hodServiceConfig, final Class<T> returnType) {
38          getContentBackend = hodServiceConfig.getRestAdapter().create(GetContentBackend.class);
39          requester = hodServiceConfig.getRequester();
40          this.returnType = returnType;
41      }
42  
43      /**
44       * Create a new GetContentServiceImpl of type {@link Documents}
45       * @param hodServiceConfig The configuration to use
46       * @return A new {@literal GetContentService<Documents>}
47       */
48      public static GetContentServiceImpl<Documents> documentsService(final HodServiceConfig hodServiceConfig) {
49          return new GetContentServiceImpl<>(hodServiceConfig, Documents.class);
50      }
51      
52      @Override
53      public T getContent(final List<String> indexReference, final ResourceIdentifier index, final GetContentRequestBuilder params) throws HodErrorException {
54          return requester.makeRequest(returnType, getBackendCaller(indexReference, index, params));
55      }
56  
57      @Override
58      public T getContent(final TokenProxy tokenProxy, final List<String> indexReference, final ResourceIdentifier index, final GetContentRequestBuilder params) throws HodErrorException {
59          return requester.makeRequest(tokenProxy, returnType, getBackendCaller(indexReference, index, params));
60      }
61  
62      private Requester.BackendCaller getBackendCaller(final List<String> indexReference, final ResourceIdentifier indexes, final GetContentRequestBuilder params) {
63          return new Requester.BackendCaller() {
64              @Override
65              public Response makeRequest(final AuthenticationToken authenticationToken) throws HodErrorException {
66                  return getContentBackend.getContent(authenticationToken, indexReference, indexes, params.build());
67              }
68          };
69      }
70  }
