27  /**
28   * Implementation of DeleteTextIndexService which polls for job completion
29   * <p/>
30   * The destroy method should be called when the service is no longer needed.
31   */
32  @Slf4j
33  public class DeleteTextIndexPollingService extends AbstractPollingService implements DeleteTextIndexService {
34  
35      private final DeleteTextIndexBackend deleteTextIndexBackend;
36      private final Requester requester;
37      private final JobService<? extends JobStatus<DeleteTextIndexResponse>> jobService;
38  
39      /**
40       * Creates a new DeleteTextIndexPollingService
41       * @param hodServiceConfig The configuration to use
42       */
43      public DeleteTextIndexPollingService(final HodServiceConfig hodServiceConfig) {
44          super();
45  
46          deleteTextIndexBackend = hodServiceConfig.getRestAdapter().create(DeleteTextIndexBackend.class);
47          requester = hodServiceConfig.getRequester();
48          jobService = new JobServiceImpl<>(hodServiceConfig, DeleteTextIndexBackend.DeleteTextIndexJobStatus.class);
49      }
50  
51      /**
52       * Creates a new DeleteTextIndexPollingService
53       * @param hodServiceConfig The configuration to use
54       * @param executorService The executor service to use while polling for status updates
55       */
56      public DeleteTextIndexPollingService(final HodServiceConfig hodServiceConfig, final ScheduledExecutorService executorService) {
57          super(executorService);
58  
59          deleteTextIndexBackend = hodServiceConfig.getRestAdapter().create(DeleteTextIndexBackend.class);
60          requester = hodServiceConfig.getRequester();
61          jobService = new JobServiceImpl<>(hodServiceConfig, DeleteTextIndexBackend.DeleteTextIndexJobStatus.class);
62      }
63  
64      @Override
65      public void deleteTextIndex(
66          final ResourceIdentifier index,
67          final HodJobCallback<DeleteTextIndexResponse> callback
68      ) throws HodErrorException {
69          final DeleteTextIndexResponse response = requester.makeRequest(DeleteTextIndexResponse.class, getInitialBackendCaller(index));
70  
71          final JobId jobId = requester.makeRequest(JobId.class, getDeletingBackendCaller(index, response));
72  
73          getExecutorService().submit(new PollingJobStatusRunnable<>(jobId, callback, getExecutorService(), jobService));
74      }
75  
76      @Override
77      public void deleteTextIndex(
78          final TokenProxy tokenProxy,
79          final ResourceIdentifier index,
80          final HodJobCallback<DeleteTextIndexResponse> callback
81      ) throws HodErrorException {
82          final DeleteTextIndexResponse response = requester.makeRequest(tokenProxy, DeleteTextIndexResponse.class, getInitialBackendCaller(index));
83  
84          final JobId jobId = requester.makeRequest(tokenProxy, JobId.class, getDeletingBackendCaller(index, response));
85  
86          getExecutorService().submit(new PollingJobStatusRunnable<>(tokenProxy, jobId, callback, getExecutorService(), jobService));
87      }
88  
89      private Requester.BackendCaller getInitialBackendCaller(final ResourceIdentifier index) {
90          return new Requester.BackendCaller() {
91              @Override
92              public Response makeRequest(final AuthenticationToken authenticationToken) throws HodErrorException {
93                  return deleteTextIndexBackend.initialDeleteTextIndex(authenticationToken, index);
94              }
95          };
96      }
97  
98      private Requester.BackendCaller getDeletingBackendCaller(final ResourceIdentifier index, final DeleteTextIndexResponse response) {
99          return new Requester.BackendCaller() {
100             @Override
101             public Response makeRequest(final AuthenticationToken authenticationToken) throws HodErrorException {
102                 return deleteTextIndexBackend.deleteTextIndex(authenticationToken, index, response.getConfirm());
103             }
104         };
105     }
