6   package com.hp.autonomy.hod.client.api.queryprofile;

8   import com.hp.autonomy.hod.client.util.MultiMap;
9   import lombok.Setter;
10  import lombok.experimental.Accessors;
11  
12  import java.util.ArrayList;
13  import java.util.Arrays;
14  import java.util.List;
15  import java.util.Map;

17  /**
18   * Helper class for building up optional parameters for the CreateQueryProfile API and UpdateQueryProfile API. The default value
19   * for all parameters is null. Null parameters will not be sent to HP Haven OnDemand
20   */
21  @Setter
22  @Accessors(chain = true)
23  public class QueryProfileRequestBuilder {
24  
25      /**
26       * @param promotionsEnabled Sets the value of the promotions_enabled parameter
27       */
28      private Boolean promotionsEnabled;
29  
30      /**
31       * @param promotionsIdentified Sets the value of the promotions_identified parameter
32       */
33      private Boolean promotionsIdentified;
34  
35      /**
36       * @param synonymsEnabled Sets the value of the synonyms_enabled parameter
37       */
38      private Boolean synonymsEnabled;
39  
40      /**
41       * @param blacklistsEnabled Sets the value of the blacklists_enabled parameter
42       */
43      private Boolean blacklistsEnabled;
44  
45      /**
46       * @param promotionCategories Sets the value of the promotion_categories parameter
47       */
48      private List<String> promotionCategories = new ArrayList<>();
49  
50      /**
51       * @param synonymCategories Sets the value of the synonym_categories parameter
52       */
53      private List<String> synonymCategories = new ArrayList<>();
54  
55      /**
56       * @param blacklistCategories Sets the value of the blacklist_categories parameter
57       */
58      private List<String> blacklistCategories = new ArrayList<>();
59  
60      /**
61       * Adds categories to the promotion_categories parameter
62       * @param category0 The first category to add
63       * @param categories Additional categories to add
64       * @return this
65       */
66      public QueryProfileRequestBuilder addPromotionCategories(final String category0, final String... categories) {
67          promotionCategories.add(category0);
68          promotionCategories.addAll(Arrays.asList(categories));
69          return this;
70      }
71  
72      /**
73       * Adds categories to the synonym_categories parameter
74       * @param category0 The first category to add
75       * @param categories Additional categories to add
76       * @return this
77       */
78      public QueryProfileRequestBuilder addSynonymCategories(final String category0, final String... categories) {
79          synonymCategories.add(category0);
80          synonymCategories.addAll(Arrays.asList(categories));
81          return this;
82      }
83  
84      /**
85       * Adds categories to the blacklist_categories parameter
86       * @param category0 The first category to add
87       * @param categories Additional categories to add
88       * @return this
89       */
90      public QueryProfileRequestBuilder addBlacklistCategories(final String category0, final String... categories) {
91          blacklistCategories.add(category0);
92          blacklistCategories.addAll(Arrays.asList(categories));
93          return this;
94      }
95  
96      /**
97       * @return A map of query parameters suitable for use with {@link QueryProfileService}. get is NOT supported on
98       * the resulting map
99       */
100     Map<String, Object> build() {
101         final Map<String, Object> params = new MultiMap<>();
102         params.put("promotions_enabled", promotionsEnabled);
103         params.put("promotions_identified", promotionsIdentified);
104         params.put("synonyms_enabled", synonymsEnabled);
105         params.put("blacklists_enabled", blacklistsEnabled);
106 
107         for (final String category : promotionCategories) {
108             params.put("promotion_categories", category);
109         }
110 
111         for (final String category : synonymCategories) {
112             params.put("synonym_categories", category);
113         }
114 
115         for (final String category : blacklistCategories) {
116             params.put("blacklist_categories", category);
117         }
118 
119         return params;
120     }
