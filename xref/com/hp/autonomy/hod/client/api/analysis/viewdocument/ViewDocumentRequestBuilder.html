6   package com.hp.autonomy.hod.client.api.analysis.viewdocument;
7   
8   import com.hp.autonomy.hod.client.util.MultiMap;
9   import lombok.AccessLevel;
10  import lombok.Setter;
11  import lombok.experimental.Accessors;
12  
13  import java.util.ArrayList;
14  import java.util.Arrays;
15  import java.util.List;
16  import java.util.Map;
17  
18  /**
19   * Helper class for building up optional parameters for the Query Text Index API. The default value for all parameters
20   * is null. Null parameters will not be sent to HP Haven OnDemand
21   */
22  @Setter
23  @Accessors(chain = true)
24  public class ViewDocumentRequestBuilder {
25  
26      private List<String> highlightExpressions = new ArrayList<>();
27      private List<String> startTags = new ArrayList<>();
28      private List<String> endTags = new ArrayList<>();
29  
30      /**
31       * Sets the value for the raw_html parameter
32       * @param rawHtml Value of the raw_html parameter
33       */
34      @Setter(AccessLevel.PACKAGE) // setting this parameter messes with return values, so should only be used with the appropriate methods
35      private Boolean rawHtml;
36  
37      /**
38       * @param highlightExpressions Sets the value of the highlight_expressions parameter
39       * @return this
40       */
41      public ViewDocumentRequestBuilder setHighlightExpressions(final List<String> highlightExpressions) {
42          this.highlightExpressions = highlightExpressions;
43          return this;
44      }
45  
46      /**
47       * Adds values to the highlight_expression parameter
48       * @param highlightExpression The first highlight expression
49       * @param highlightExpressions The remaining highlight expressions
50       * @return this
51       */
52      public ViewDocumentRequestBuilder addHighlightExpressions(final String highlightExpression, final String... highlightExpressions) {
53          this.highlightExpressions.add(highlightExpression);
54          this.highlightExpressions.addAll(Arrays.asList(highlightExpressions));
55          return this;
56      }
57  
58      /**
59       * @param startTags Sets the value of the start_tag parameter
60       * @return this
61       */
62      public ViewDocumentRequestBuilder setStartTags(final List<String> startTags) {
63          this.startTags = startTags;
64          return this;
65      }
66  
67      /**
68       * Adds values to the start_tag parameter
69       * @param startTag The first start tag
70       * @param startTags The remaining start tags
71       * @return this
72       */
73      public ViewDocumentRequestBuilder addStartTags(final String startTag, final String... startTags) {
74          this.startTags.add(startTag);
75          this.startTags.addAll(Arrays.asList(startTags));
76          return this;
77      }
78  
79      /**
80       * @param endTags Sets the value of the end_tag parameter
81       * @return this
82       */
83      public ViewDocumentRequestBuilder setEndTags(final List<String> endTags) {
84          this.endTags = endTags;
85          return this;
86      }
87  
88      /**
89       * Adds values to the end_tag parameter
90       * @param endTag The first end tag
91       * @param endTags The remaining end tags
92       * @return this
93       */
94      public ViewDocumentRequestBuilder addEndTags(final String endTag, final String... endTags) {
95          this.endTags.add(endTag);
96          this.endTags.addAll(Arrays.asList(endTags));
97          return this;
98      }
99  
100     /**
101      * @return A map of query parameters suitable for use with {@link ViewDocumentBackend}. get is NOT supported on
102      * the resulting map
103      */
104     Map<String, Object> build() {
105         final Map<String, Object> params = new MultiMap<>();
106 
107         params.put("raw_html", rawHtml);
108 
109         for (final String highlightExpression : highlightExpressions) {
110             params.put("highlight_expression", highlightExpression);
111         }
112 
113         for (final String startTag : startTags) {
114             params.put("start_tag", startTag);
115         }
116 
117         for (final String endTag : endTags) {
118             params.put("end_tag", endTag);
119         }
120 
121         return params;
122     }

124 }
