25      protected static final String DOMAIN_NAME = "IOD-TEST-DOMAIN";

30      private HodServiceConfig hodServiceConfig;
31      private RestAdapter restAdapter;
32      private AuthenticationToken token;
33      private TokenProxy tokenProxy;
34  
35      public void setUp() {
36          hodServiceConfig = HodServiceConfigFactory.getHodServiceConfig(null, endpoint);
37          restAdapter = hodServiceConfig.getRestAdapter();
38  
39          final AuthenticationBackend authenticationBackend = restAdapter.create(AuthenticationBackend.class);
40  
41          try {
42              token = authenticationBackend.authenticateApplication(
43                  new ApiKey(System.getProperty("hp.dev.placeholder.hod.apiKey")),
44                      APPLICATION_NAME,
45                      DOMAIN_NAME,
46                  TokenType.simple
47              ).getToken();
48  
49              tokenProxy = hodServiceConfig.getTokenRepository().insert(token);
50          } catch (final HodErrorException | IOException e) {
51              throw new AssertionError("COULD NOT OBTAIN TOKEN");
52          }
53      }
