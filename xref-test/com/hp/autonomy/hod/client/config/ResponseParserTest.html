33  public class ResponseParserTest {
34  
35      private ResponseParser responseParser;
36      private Converter converter;
37      private TokenRepository tokenRepository;
38  
39      @Before
40      public void setUp() {
41          converter = mock(Converter.class);
42          tokenRepository = mock(TokenRepository.class);
43  
44          responseParser = new ResponseParser(tokenRepository, converter);
45      }
46  
47      @Test
48      public void testParseResponseWithARefreshedToken() throws IOException, ConversionException {
49          final AuthenticationToken newToken = mock(AuthenticationToken.class);
50  
51          final List<Header> headers = Arrays.asList(
52              new Header("foo", "bar"),
53              new Header("refresh_token", "TOKEN") // the converter isn't real so the value can be anything
54          );
55  
56          final Object expectedReturnValue = new Object();
57  
58          // can't mock this because it is final
59          final Response response = new Response("", 200, "", headers, mock(TypedInput.class));
60  
61          when(converter.fromBody(isA(TypedInput.class), eq(AuthenticationToken.class))).thenReturn(newToken);
62          when(converter.fromBody(isA(TypedInput.class), eq(Object.class))).thenReturn(expectedReturnValue);
63  
64          final TokenProxy tokenProxy = new TokenProxy();
65  
66          final Object returnValue = responseParser.parseResponse(tokenProxy, Object.class, response);
67  
68          verify(tokenRepository).update(tokenProxy, newToken);
69          assertThat(returnValue, is(expectedReturnValue));
70      }
71  
72      @Test
73      public void testResponseParserWithoutARefreshedToken() throws ConversionException, IOException {
74          final Object expectedReturnValue = new Object();
75  
76          // can't mock this because it is final
77          final Response response = new Response("", 200, "", Collections.<Header>emptyList(), mock(TypedInput.class));
78
