119     @Test
120     public void sso() throws HodErrorException, IOException, URISyntaxException {
121         final String origin = "http://example.com";
122         final Collection<String> allowedOrigins = Arrays.asList("https://another-example.com", origin);
123 
124         // Use an HttpClient to represent the browser in the SSO process
125         final HttpClient browser = createBrowser();
126 
127         // Authenticate the user, creating a cookie in the browser on the HOD endpoint domain (replaces the SSO page)
128         // TODO: The API key used to authenticate the user should be configurable
129         authenticateUserUnbound(browser, apiKey);
130 
131         // Authenticate the application
132         // TODO: The API key used to authenticate the application should be configurable
133         final AuthenticationToken unboundToken = authenticationService.authenticateUnbound(apiKey);
134 
135         // Get a list of applications and users which match the application and user authentications by executing a
136         // signed request in the browser
137         final SignedRequest getApplicationRequest = authenticationService.combinedGetRequest(allowedOrigins, unboundToken);
138         final List<ApplicationAndUsers> applicationResponse = makeSignedRequest(browser, getApplicationRequest, origin, GET_APPLICATION_RESPONSE_REFERENCE);
139 
140         assertThat(applicationResponse, not(empty()));
141 
142         final ApplicationAndUsers applicationAndUsers = applicationResponse.get(0);
143         assertThat(applicationAndUsers.domain, notNullValue());
144         assertThat(applicationAndUsers.name, notNullValue());
145         assertThat(applicationAndUsers.users, not(empty()));
146 
147         final ApplicationAndUsers.User user = applicationAndUsers.users.get(0);
148         assertThat(user.storeDomain, notNullValue());
149         assertThat(user.storeName, notNullValue());
150 
151         // Sign a request to obtain a combined token from the browser
152         final SignedRequest combinedRequest = authenticationService.combinedRequest(
153                 allowedOrigins,
154                 unboundToken,
155                 applicationAndUsers.domain,
156                 applicationAndUsers.name,
157                 user.storeDomain,
158                 user.storeName,
159                 TokenType.simple,
160                 true
161         );
162 
163         // Get the combined token
164         final AuthenticationTokenResponse authenticationTokenResponse = makeSignedRequest(browser, combinedRequest, origin, TOKEN_RESPONSE_REFERENCE);
165         final AuthenticationToken combinedToken = authenticationTokenResponse.getToken();
166 
167         assertThat(combinedToken, notNullValue());
168     }

182     private void authenticateUserUnbound(final HttpClient browser, final ApiKey apiKey) throws URISyntaxException, IOException {
183         final URI uri = new URIBuilder(endpointUrl + "/2/authenticate/unbound")
184                 .addParameter("apiKey", apiKey.toString())
185                 .build();
186 
187         final HttpPost postRequest = new HttpPost(uri);
188         postRequest.setHeader("Content-Type", "application-x-www-form-urlencoded");
189 
190         final List<NameValuePair> bodyPairs = new LinkedList<>();
191         bodyPairs.add(new BasicNameValuePair("enable_sso", "true"));
192         bodyPairs.add(new BasicNameValuePair("token_type", TokenType.simple.toString()));
193         final String bodyString = URLEncodedUtils.format(bodyPairs, StandardCharsets.UTF_8);
194 
195         final HttpEntity body = new ByteArrayEntity(bodyString.getBytes(StandardCharsets.UTF_8));
196         postRequest.setEntity(body);
197 
198         makeRequest(browser, postRequest, TOKEN_RESPONSE_REFERENCE);
199     }

201     private <T> T makeSignedRequest(final HttpClient browser, final SignedRequest request, final String origin, final TypeReference<T> responseType) throws IOException {
202         final RequestBuilder uriRequestBuilder = RequestBuilder.create(request.getVerb().toString())
203                 .addHeader("token", request.getToken())
204                 .addHeader("Origin", origin)
205                 .setUri(request.getUrl());
206 
207         if (request.getBody() != null) {
208             final HttpEntity body = new ByteArrayEntity(request.getBody().getBytes(StandardCharsets.UTF_8));
209             uriRequestBuilder.setEntity(body);
210             uriRequestBuilder.addHeader("Content-Type", "application/x-www-form-urlencoded");
211         }
212 
213         return makeRequest(browser, uriRequestBuilder.build(), responseType);
214     }
215 
216     private <T> T makeRequest(final HttpClient browser, final HttpUriRequest request, final TypeReference<T> responseType) throws IOException {
217         final HttpResponse response = browser.execute(request);
218         final int statusCode = response.getStatusLine().getStatusCode();
219 
220         assertThat(statusCode, is(200));
221 
222         try(InputStream content = response.getEntity().getContent()){
223             return mapper.readValue(content, responseType);
224         }
225     }
